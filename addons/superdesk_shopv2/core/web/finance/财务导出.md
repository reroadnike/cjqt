SELECT
    o.*,
    
    ofi.id AS finance_id,
    ofi.create_invoice_time,
    ofi.invoice_sn,
    
    core_enterprise.name AS enterprise_name,
    
    po.ordersn AS parent_ordersn,
    po.price AS parent_price,
    
    mu.merchname,
    
    m.nickname,
    m.id AS MID,
    m.realname AS mrealname,
    m.mobile AS mmobile,
    m.core_enterprise,
    
    a.realname AS arealname,
    a.mobile AS amobile,
    a.province AS aprovince,
    a.city AS acity,
    a.area AS aarea,
    a.town AS atown,
    a.address AS aaddress
FROM
    `ims_superdesk_shop_order_finance` ofi
LEFT JOIN `ims_superdesk_shop_order` o ON
    o.id = ofi.orderid
LEFT JOIN `ims_superdesk_shop_order` po ON
    po.id = o.parentid
LEFT JOIN `ims_superdesk_shop_merch_user` mu ON
    mu.id = o.merchid
LEFT JOIN `ims_superdesk_shop_member` m ON
    m.openid = o.openid
LEFT JOIN `ims_superdesk_shop_member_address` a ON
    a.id = o.addressid
LEFT JOIN `ims_superdesk_core_virtualarchitecture` core_enterprise ON
    core_enterprise.id = m.core_enterprise
WHERE
    o.uniacid = 16
ORDER BY
    m.core_enterprise ASC,o.createtime ASC
